# Program

````c
#include <stdio.h>
#include <stdlib.h>

int main()
{
    long long n = 0;
    int magic_count = 0;
    scanf("%lld", &n);
    while (n)
    {
	if (n % 10 == 4 || n % 10 == 7) magic_count++;
	n /= 10;
    }
    printf("%s\n", ((magic_count == 4 || magic_count == 7)?"YES":"NO"));
    return 0;
}
````

````cpp
#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    string n;
    int magic_count = 0;
    cin >> n;
    for (auto d : n)
	if (d == '4' || d == '7') magic_count++;
    cout << ((magic_count == 4 || magic_count == 7)?"YES":"NO") << '\n';
    return 0;
}
````
